CREATE TABLE CUSTOMER_TABLE (
    CUSTOMER_ID VARCHAR(15) NOT NULL PRIMARY KEY,
    FIRST_NAME VARCHAR(20) NOT NULL,
    MIDDLE_NAME VARCHAR(20) NOT NULL,
    LAST_NAME VARCHAR(20) NOT NULL,
    BIRTH_DATE DATE NOT NULL,
    AGE INT NOT NULL,
    GENDER VARCHAR(10)
);

CREATE TABLE CUSTOMER_ADDRESS_TABLE (
    CUSTOMER_ID VARCHAR(15) NOT NULL,
    STREET VARCHAR(20),
    BARANGAY VARCHAR(20),
    CITY VARCHAR(20),
    PROVINCE VARCHAR(20),
    FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER_TABLE(CUSTOMER_ID)
);

CREATE TABLE CUSTOMER_ACCOUNT_TABLE (
    CUSTOMER_ID VARCHAR(15) NOT NULL,
    USERNAME VARCHAR(20),
    PASSWORD VARCHAR(150),
    FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER_TABLE(CUSTOMER_ID)
);

CREATE TABLE PRODUCT_TABLE(
    PRODUCT_ID VARCHAR(15) NOT NULL PRIMARY KEY,
    PRODUCT_NAME VARCHAR(30),
    PRODUCT_PRICE DECIMAL(9, 2),
    PRODUCT_IMG VARCHAR(30),
    PRODUCT_STOCKS INT
);

CREATE TABLE CART_TABLE(
    CUSTOMER_ID VARCHAR(15) NOT NULL,
    PRODUCT_ID VARCHAR(15),
    QUANTITY INT,
    FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER_TABLE(CUSTOMER_ID),
    FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCT_TABLE(PRODUCT_ID)
);
